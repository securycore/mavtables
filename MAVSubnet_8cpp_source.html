152 MAVSubnet::MAVSubnet(std::string subnet)
153  : address_(0)
154 {
155  // Extract parts of subnet string.
156  std::vector<std::string> parts;
157  boost::char_separator<char> sep("", ":/\\");
158  boost::tokenizer<boost::char_separator<char>> tokens(subnet, sep);
159 
160  // std::cout << "subnet:" << subnet << std::endl;
161  for (auto i : tokens)
162  {
163  parts.push_back(i);
164  }
165 
166  // Ensure proper format.
167  if (parts.size() != 3)
168  {
169  throw std::invalid_argument("Invalid MAVLink subnet: \"" + subnet + "\"");
170  }
171 
172  address_ = MAVAddress(parts[0]);
173 
174  // Determine format of subnet string.
175  unsigned int slashmask;
176  switch (parts[1].at(0))
177  {
178  case ':':
179  mask_ = MAVAddress(parts[2]).address();
180  break;
181 
182  case '/':
183  std::istringstream(parts.at(2)) >> slashmask;
184 
185  if (slashmask > 16)
186  {
187  throw std::out_of_range(
188  "forward slash mask ("
189  + std::to_string(slashmask)
190  + ") is outside of allowed range (0 - 16).");
191  }
192 
193  mask_ = (0xFFFF << (16 - slashmask)) & 0xFFFF;
194  break;
195 
196  case '\\':
197  std::istringstream(parts.at(2)) >> slashmask;
198 
199  if (slashmask > 8)
200  {
201  throw std::out_of_range(
202  "backslash mask ("
203  + std::to_string(slashmask)
204  + ") is outside of allowed range (0 - 8).");
205  }
206 
207  mask_ = (0xFFFF << (8 - slashmask)) & 0x00FF;
208  break;
209  }
210 }
