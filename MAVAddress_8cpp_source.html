97 MAVAddress::MAVAddress(std::string address)
98 {
99  // Check validity of address string.
100  if (address.size() < 3 || !(isdigit(address.front()))
101  || !isdigit(address.back()))
102  {
103  throw std::invalid_argument("Invalid MAVLink address string.");
104  }
105 
106  for (auto c : address)
107  {
108  if (!(c == '.' || isdigit(c)))
109  {
110  throw std::invalid_argument("Invalid MAVLink address string.");
111  }
112  }
113 
114  // Read address string.
115  std::replace(address.begin(), address.end(), '.', ' ');
116  std::vector<unsigned int> octets;
117  std::istringstream ss(address);
118  unsigned int i;
119 
120  while (ss >> i)
121  {
122  octets.push_back(i);
123  }
124 
125  // Check for correct number of octets.
126  if (octets.size() != 2 || !ss.eof())
127  {
128  throw std::invalid_argument("Invalid MAVLink address string.");
129  }
130 
131  construct_(octets[0], octets[1]);
132 }
