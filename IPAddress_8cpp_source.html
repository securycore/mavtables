48 {
49  if (address > 0xFFFFFFFF)
50  {

53  ") is outside of the allowed range (0x00000000 - 0xffffffff).";
54  throw std::out_of_range(ss.str());

56

60  ") is outside of the allowed range (0 - 65535).");

113 IPAddress::IPAddress(std::string address)
114 {
115  // Separate port from address.
116  unsigned int port = 0;
117  std::vector<std::string> parts;
118  boost::split(parts, address, [](char c)
119  {
120  return c == ':';
121  });
122 
123  // Read port.
124  if (parts.size() == 2)
125  {
126  std::istringstream(parts.back()) >> port;
127  parts.pop_back();
128  }
129 
130  if (parts.size() != 1)
131  {
132  throw std::invalid_argument("Invalid IP address string.");
133  }
134 
135  address = parts.back();
136 
137  // Check validity of address string.
138  if (address.size() < 7 || !(isdigit(address.front()))
139  || !isdigit(address.back()))
140  {
141  throw std::invalid_argument("Invalid IP address string.");
142  }
143 
144  for (auto c : address)
145  {
146  if (!(c == '.' || isdigit(c)))
147  {
148  throw std::invalid_argument("Invalid IP address string.");
149  }
150  }
151 
152  // Read address.
153  std::replace(address.begin(), address.end(), '.', ' ');
154  std::vector<unsigned long> octets;
155  std::istringstream ss(address);
156  unsigned long octet;
157 
158  while (ss >> octet)
159  {
160  octets.push_back(octet);
161  }
162 
163  // Ensure proper number of octets.
164  if (octets.size() != 4)
165  {
166  throw std::invalid_argument("Invalid IP address string.");
167  }
168 
169  // Ensure octets are between 0 and 255.
170  for (auto i : octets)
171  if (i > 255)
172  {
173  throw std::out_of_range("address octet (" + std::to_string(i) +
174  ") is outside of the allowed range (0 - 255).");
175  }
176 
177  {
178  }
179 
180  construct_((octets[0] << 8 * 3) | (octets[1] << 8 * 2) |
181  (octets[2] << 8) | octets[3], port);
182 }

353 /* Lookup an IP address based on a hostname.
354  *
355  * This version is unix only and will be called by \ref dnslookup on UNIX
356  * systems.
357  */
358 static IPAddress unix_dnslookup(const std::string &url, unsigned int port)
359 {
360  std::set<IPAddress> addresses;
361  // Setup hints.
362  struct addrinfo hints;
363  std::memset(&hints, 0, sizeof(hints));
364  hints.ai_family = AF_INET; // IPv4 only (for now)
365  hints.ai_protocol = static_cast<int>(port);
366  // Get IP addresses.
367  struct addrinfo *result_ptr = nullptr;
368 
369  if (getaddrinfo(url.c_str(), nullptr, &hints, &result_ptr))
370  {
371  throw DNSLookupError(url);
372  }
373 
374  std::unique_ptr<struct addrinfo, void(*)(struct addrinfo *)> result(result_ptr,
375  freeaddrinfo);
376  result_ptr = nullptr;
377 
378  for (result_ptr = result.get(); result_ptr != nullptr;
379  result_ptr = result_ptr->ai_next)
380  {
381  struct sockaddr_in *address_ptr = reinterpret_cast<struct sockaddr_in *>
382  (result_ptr->ai_addr);
383  unsigned long address = ntohl(address_ptr->sin_addr.s_addr);
384  // unsigned int port_ = ntohs(address_ptr->sin_port);
385  addresses.insert(IPAddress(address, port));
386  }
387 
388  // This should never be true but it's here just in case.
389  if (addresses.empty())
390  {
391  throw DNSLookupError(url);
392  }
393 
394  return *(addresses.begin());
395 }
